import { Body, Controller, Get, HttpCode, Param, Post, Query } from '@nestjs/common';
import { type Organization, OrganizationHeaders } from 'src/common/decorators/organization.decorator';
import { Strategies, Strategy } from 'src/common/decorators/strategy.decorator';
 
import { BasketActions } from '../common/basket.enum';
import { BasketStrategyContextV1 } from './basket-strategy.context';
import { CreateBasketRequestBodyDTOV1 } from './dto/request/create-basket-request.dto';
import { LockBasketRequestPathParamsDTOV1, LockBasketRequestQueryParamsDTOV1 } from './dto/request/lock-basket-request.dto';
import { PayBasketRequestBodyDTOV1, PayBasketRequestPathParamsDTOV1 } from './dto/request/pay-basket-request.dto';
import { UnLockBasketRequestPathParamsDTOV1 } from './dto/request/unlock-basket-request.dto';
 
@Controller({ version: '1', path: 'baskets' })
export class BasketControllerV1 {
  constructor(private readonly basketStrategyContextV1: BasketStrategyContextV1) {}
 
  @Post()
  create(@Strategy() strategy: Strategies, @OrganizationHeaders() organization: Organization, @Body() body: CreateBasketRequestBodyDTOV1) {
    return this.basketStrategyContextV1.findStrategy(strategy).handle(BasketActions.CREATE, { organization, body });
  }
 
  @Get()
  findAll(@Strategy() strategy: Strategies, @OrganizationHeaders() organization: Organization) {
    return this.basketStrategyContextV1.findStrategy(strategy).handle(BasketActions.FIND_ALL, { organization });
  }
 
  @Post('/:id/lock')
  @HttpCode(200)
  lock(@Strategy() strategy: Strategies, @OrganizationHeaders() organization: Organization, @Param() pathParams: LockBasketRequestPathParamsDTOV1, @Query() queryParams: LockBasketRequestQueryParamsDTOV1) {
    return this.basketStrategyContextV1.findStrategy(strategy).handle(BasketActions.LOCK, { organization, pathParams, queryParams });
  }
 
  @Post('/:id/unlock')
  @HttpCode(200)
  unlock(@Strategy() strategy: Strategies, @OrganizationHeaders() organization: Organization, @Param() pathParams: UnLockBasketRequestPathParamsDTOV1) {
    return this.basketStrategyContextV1.findStrategy(strategy).handle(BasketActions.UNLOCK, { organization, pathParams });
  }
 
  @Post('/:id/pay')
  @HttpCode(200)
  pay(@Strategy() strategy: Strategies, @OrganizationHeaders() organization: Organization, @Body() body: PayBasketRequestBodyDTOV1, @Param() pathParams: PayBasketRequestPathParamsDTOV1) {
    return this.basketStrategyContextV1.findStrategy(strategy).handle(BasketActions.PAY, { organization, body, pathParams });
  }
}
